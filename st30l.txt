sudo cp st30l.service /etc/systemd/system/st30l.service
sudo systemctl daemon-reload
sudo systemctl enable st30l.service
sudo systemctl start st30l.service
sudo systemctl status st30l.service

# Create the directory for the share

mkdir /home/haas/Haas/st30l

# Create the share configuration

[st30l]
    comment = Logger for ST30L
    path = /home/haas/Haas/st30l
    read only = no
    browsable = yes
    writable = yes
    public = no
    valid users = @HaasGroup, haas # Ensure the user is valid
    force user = haas
    force group = HaasGroup
    create mask = 0664
    force create mode = 0664
    directory mask = 0775
    force directory mode = 0775
